## Concepts clés

### Dessiner sur l'écran

```python
from machine import Pin, SPI
import ili9341

spi = SPI(1, baudrate=40_000_000, sck=Pin(12), mosi=Pin(13))
lcd = ili9341.ILI9341(spi, cs=Pin(11), dc=Pin(10), rst=Pin(14))

# Couleurs en RGB565
BLANC  = 0xFFFF
NOIR   = 0x0000
ROUGE  = 0xF800
VERT   = 0x07E0
BLEU   = 0x001F

# Effacer l'écran
lcd.fill(NOIR)

# Dessiner des formes
lcd.fill_rect(10, 10, 100, 50, BLEU)    # rectangle plein
lcd.rect(10, 10, 100, 50, BLANC)        # contour blanc
lcd.text("Bonjour!", 20, 30, BLANC)     # texte
```

### Lire le tactile

```python
from machine import Pin, SPI
import xpt2046

spi_touch = SPI(2, baudrate=1_000_000, sck=Pin(12), mosi=Pin(13), miso=Pin(14))
touch = xpt2046.XPT2046(spi_touch, cs=Pin(16), int_pin=Pin(17))

while True:
    if touch.is_pressed():
        x, y = touch.get_position()
        print(f"Touché : x={x}, y={y}")
```

### Bouton virtuel

```python
class Bouton:
    def __init__(self, lcd, x, y, w, h, texte, couleur=0x001F):
        self.lcd = lcd
        self.x, self.y, self.w, self.h = x, y, w, h
        self.texte = texte
        self.couleur = couleur
        self.dessiner()

    def dessiner(self, appuye=False):
        c = 0x4208 if appuye else self.couleur
        self.lcd.fill_rect(self.x, self.y, self.w, self.h, c)
        self.lcd.rect(self.x, self.y, self.w, self.h, 0xFFFF)
        tx = self.x + (self.w - len(self.texte) * 8) // 2
        ty = self.y + (self.h - 8) // 2
        self.lcd.text(self.texte, tx, ty, 0xFFFF)

    def touche(self, x, y):
        return self.x <= x <= self.x + self.w and self.y <= y <= self.y + self.h
```
